Theorem andb_true_elim2 : forall b c : bool,
  andb b c = true -> c = true.
Proof.
  intros b c.
  destruct b eqn:Eb.
  - simpl. intros H. rewrite -> H. reflexivity.
  - simpl. destruct c eqn:Ec.
    + intros H. reflexivity.
    + intros H. rewrite -> H. reflexivity.
Qed.
